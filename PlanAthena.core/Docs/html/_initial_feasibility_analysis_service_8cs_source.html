21 public Task<List<MessageValidationDto>> AnalyserFaisabiliteAsync(Chantier chantier)
22 {
23 var messages = new List<MessageValidationDto>();
24
25 // 1. Lister tous les MetierId distincts requis par les tâches.
26 // L'utilisation d'un HashSet est efficace pour garantir l'unicité et pour des recherches rapides.
27 var metiersRequis = chantier.ObtenirToutesLesTaches()
28 .Select(tache => tache.MetierRequisId)
29 .ToHashSet();
30
31 // Si aucune tâche ne requiert de métier, il n'y a rien à analyser.
32 if (metiersRequis.Count == 0)
33 {
34 return Task.FromResult(messages);
35 }
36
37 // 2. Lister tous les MetierId distincts possédés par les ouvriers.
38 var metiersDisponibles = chantier.Ouvriers.Values
39 .SelectMany(ouvrier => ouvrier.Competences.Keys)
40 .ToHashSet();
41
42 // 3. Identifier les métiers requis qui ne sont pas disponibles.
43 // La méthode Except() est parfaite et très lisible pour cette opération d'ensemble.
44 var metiersNonCouverts = metiersRequis.Except(metiersDisponibles);
45
46 // 4. Créer un message d'avertissement pour chaque métier non couvert.
47 foreach (var metierId in metiersNonCouverts)
48 {
49 // On récupère le nom du métier pour un message plus clair pour l'utilisateur.
50 string nomMetier = chantier.Metiers.TryGetValue(metierId, out var metier) ? metier.Nom : "Inconnu";
51
52 messages.Add(new MessageValidationDto
53 {
54 Type = TypeMessageValidation.Avertissement,
55 CodeMessage = METIER_NON_COUVERT_CODE,
56 Message = $"Le métier '{nomMetier}' ({metierId.Value}) est requis par une ou plusieurs tâches, mais aucun ouvrier ne possède cette compétence.",
57 ProprieteConcernee = "Ouvriers" // La propriété impactée est la liste des ouvriers.
58 });
59 }
60
61 return Task.FromResult(messages);
62 }
63 }
