103 public TacheDtoValidator()

106 RuleFor(x => x.Nom).NotEmpty().MaximumLength(250);
107 RuleFor(x => x.BlocId).NotEmpty();

110
111 RuleFor(x => x.Dependencies)

114 .Must(HaveUniqueValues)
115 .WithMessage("Les dépendances de tâches doivent être uniques pour une même tâche.");

117 private bool NotContainSelfReference(string id, IReadOnlyList<string> dependancyIds) =>
