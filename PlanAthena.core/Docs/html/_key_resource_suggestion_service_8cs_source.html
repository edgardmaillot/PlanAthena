20 public Task<List<OuvrierId>> SuggererOuvriersClesAsync(Chantier chantier)
21 {
22 if (chantier.Ouvriers.Count == 0 || !chantier.ObtenirToutesLesTaches().Any())
23 {
24 return Task.FromResult(new List<OuvrierId>());
25 }
26
27 // 1. Calculer la charge totale en heures-homme pour chaque métier.
28 var chargeParMetier = chantier.ObtenirToutesLesTaches()
29 .GroupBy(t => t.MetierRequisId)
30 .ToDictionary(g => g.Key, g => g.Sum(t => t.HeuresHommeEstimees.Value));
31
32 // 2. Calculer la rareté de chaque métier (combien d'ouvriers le possèdent).
33 var rareteParMetier = chantier.Ouvriers.Values
34 .SelectMany(o => o.Competences.Keys)
35 .GroupBy(metierId => metierId)
36 .ToDictionary(g => g.Key, g => g.Count());
37
38 // 3. Calculer un score de criticité pour chaque ouvrier.
39 var scoresOuvriers = new Dictionary<OuvrierId, double>();
40 foreach (var ouvrier in chantier.Ouvriers.Values)
41 {
42 double scoreTotal = 0;
43 foreach (var competenceId in ouvrier.Competences.Keys)
44 {
45 chargeParMetier.TryGetValue(competenceId, out int charge);
46 rareteParMetier.TryGetValue(competenceId, out int rarete);
47
48 if (charge > 0 && rarete > 0)
49 {
50 // Le score pour cette compétence est la charge de travail divisée par le nombre de personnes pouvant la faire.
51 // Une charge élevée et une rareté faible (ex: 1 seule personne) donnent un score élevé.
52 double scoreCompetence = (double)charge / rarete;
53 scoreTotal += scoreCompetence;
54 }
55 }
56 scoresOuvriers[ouvrier.Id] = scoreTotal;
57 }
58
59 // 4. Trier les ouvriers par score et retourner les meilleurs suggestions.
60 var ouvriersDejaCles = chantier.ConfigCdC?.OuvriersClefs ?? new HashSet<OuvrierId>();
61
62 var suggestions = scoresOuvriers
63 .Where(kvp => !ouvriersDejaCles.Contains(kvp.Key)) // Exclure ceux déjà marqués comme clés
64 .OrderByDescending(kvp => kvp.Value) // Trier par score décroissant
65 .Select(kvp => kvp.Key) // Sélectionner l'ID de l'ouvrier
66 .Take(MAX_SUGGESTIONS) // Prendre les N meilleurs
67 .ToList();
68
69 return Task.FromResult(suggestions);
70 }
71 }
