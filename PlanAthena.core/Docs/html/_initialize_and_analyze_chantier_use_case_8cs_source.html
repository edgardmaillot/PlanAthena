37 public async Task<ChantierSetupAnalysisResultDto> ExecuteAsync(ChantierSetupInputDto inputDto)
38 {
39 // --- Phase 1: Validation des Entrées ---
40 var allMessages = new List<MessageValidationDto>();
41
42 // 1a. Validation structurelle (FluentValidation)
43 var fluentValidationResult = await _fluentValidator.ValidateAsync(inputDto);
44 if (!fluentValidationResult.IsValid)
45 {
46 allMessages.AddRange(fluentValidationResult.Errors.Select(e => new MessageValidationDto
47 {
48 Type = TypeMessageValidation.Erreur,
49 CodeMessage = e.ErrorCode ?? "ERR_FLUENT_VALIDATION",
50 Message = e.ErrorMessage,
51 ProprieteConcernee = e.PropertyName
52 }));
53 return BuildResult(inputDto.ChantierId, EtatTraitementInput.EchecValidation, allMessages, null);
54 }
55
56 // 1b. Mapping et validation des invariants du domaine
57 var (chantier, mappingMessages) = await _inputMapper.MapToDomainAsync(inputDto);
58 allMessages.AddRange(mappingMessages);
59 if (chantier == null)
60 {
61 return BuildResult(inputDto.ChantierId, EtatTraitementInput.EchecValidation, allMessages, null);
62 }
63
64 // 1c. Validation de la logique (cycles)
65 var cycleMessages = await _chantierValidationService.ValiderChantierCompletAsync(inputDto, chantier);
66 allMessages.AddRange(cycleMessages);
67
68 // Point de contrôle final de la validation
69 if (allMessages.Any(m => m.Type == TypeMessageValidation.Erreur))
70 {
71 return BuildResult(chantier.Id.Value, EtatTraitementInput.EchecValidation, allMessages, null);
72 }
73
74 // --- Phase 2: Analyse (uniquement si la validation est réussie) ---
75
76 // 2a. Analyse de faisabilité
77 var feasibilityMessages = await _feasibilityService.AnalyserFaisabiliteAsync(chantier);
78 allMessages.AddRange(feasibilityMessages);
79
80 // 2b. Suggestion des ressources clés
81 var suggestedKeyResourceIds = await _keyResourceService.SuggererOuvriersClesAsync(chantier);
82 var analyseRessourcesResult = new AnalyseRessourcesResultatDto
83 {
84 OuvriersClesSuggereIds = suggestedKeyResourceIds.Select(id => id.Value).ToList()
85 };
86 if (suggestedKeyResourceIds.Any())
87 {
88 allMessages.Add(new MessageValidationDto
89 {
90 Type = TypeMessageValidation.Suggestion,
91 CodeMessage = "SUGGEST_KEY_RESOURCES",
92 Message = $"Suggère les ouvriers suivants comme clés : {string.Join(", ", suggestedKeyResourceIds.Select(id => id.Value))}"
93 });
94 }
95
96 // --- Phase 3: Construction du Résultat Final ---
97 var finalEtat = allMessages.Any(m => m.Type == TypeMessageValidation.Avertissement)
98 ? EtatTraitementInput.SuccesAvecAvertissements
99 : EtatTraitementInput.Succes;
100
101 return BuildResult(chantier.Id.Value, finalEtat, allMessages, analyseRessourcesResult);
102 }
