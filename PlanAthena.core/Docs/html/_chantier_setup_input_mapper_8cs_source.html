25 public Task<(Chantier? chantier, List<MessageValidationDto> messages)> MapToDomainAsync(ChantierSetupInputDto inputDto)
26 {
27 var messages = new List<MessageValidationDto>();
28 Chantier? chantierDomaine = null;
29
30 try
31 {
32 // 1. Mapper les VOs simples du Chantier
33 var chantierId = new ChantierId(inputDto.ChantierId); // Peut lever une exception si inputDto.ChantierId est invalide
34 var periodeSouhaitee = new PeriodePlanification(
35 inputDto.DateDebutSouhaitee,
36 inputDto.DateFinSouhaitee,
37 inputDto.FlexibiliteDebut,
38 inputDto.FlexibiliteFin);
39
40 // 2. Créer le CalendrierOuvreChantier via le service
41 // Ce service encapsulera la logique NodaTime et la création du VO complexe.
42 // Pour l'instant, ICalendrierService n'est pas encore implémenté,
43 // donc cette ligne est conceptuelle.
44 CalendrierOuvreChantier calendrierOuvre = _calendrierService.CreerCalendrierOuvreChantier(
45 inputDto.CalendrierTravail,
46 inputDto.DateDebutSouhaitee,
47 inputDto.DateFinSouhaitee);
48
49 // 3. Mapper Metiers
50 var metiersDomaine = new List<Metier>();
51 foreach (var metierDto in inputDto.Metiers)
52 {
53 var metierId = new MetierId(metierDto.MetierId);
54 var prerequisIds = metierDto.PrerequisMetierIds.Select(id => new MetierId(id)).ToList();
55 metiersDomaine.Add(new Metier(metierId, metierDto.Nom, prerequisIds));
56 }
57
58 // 4. Mapper Ouvriers et leurs Compétences
59 var ouvriersDomaine = new List<Ouvrier>();
60 foreach (var ouvrierDto in inputDto.Ouvriers)
61 {
62 var ouvrierId = new OuvrierId(ouvrierDto.OuvrierId);
63 var coutJournalier = new CoutJournalier(ouvrierDto.CoutJournalier); // Devise par défaut EUR
64
65 var competencesDomaine = new List<Competence>();
66 foreach (var compDto in ouvrierDto.Competences)
67 {
68 competencesDomaine.Add(new Competence(
69 new MetierId(compDto.MetierId),
70 compDto.Niveau,
71 compDto.PerformancePct));
72 }
73 ouvriersDomaine.Add(new Ouvrier(ouvrierId, ouvrierDto.Nom, ouvrierDto.Prenom, coutJournalier, competencesDomaine));
74 }
75
76 // 5. Mapper Blocs et leurs Taches
77 // Il faut d'abord mapper toutes les tâches d'un bloc, puis créer le bloc.
78 var blocsDomaine = new List<BlocTravail>();
79 foreach (var blocDto in inputDto.Blocs)
80 {
81 var blocId = new BlocId(blocDto.BlocId);
82 var capacite = new CapaciteOuvriers(blocDto.CapaciteMaxOuvriers);
83
84 var tachesPourCeBloc = new List<Tache>();
85 // Filtrer les TacheDto qui appartiennent à ce bloc
86 var tachesDtoPourCeBloc = inputDto.Taches.Where(t => t.BlocId == blocDto.BlocId);
87
88 foreach (var tacheDto in tachesDtoPourCeBloc)
89 {
90 var tacheId = new TacheId(tacheDto.TacheId);
91 var duree = new DureeHeuresHomme(tacheDto.HeuresHommeEstimees);
92 var metierRequisId = new MetierId(tacheDto.MetierId);
93 var dependances = tacheDto.Dependencies.Select(id => new TacheId(id)).ToList();
94
95 tachesPourCeBloc.Add(new Tache(tacheId, tacheDto.Nom, blocId, duree, metierRequisId, dependances));
96 }
97 blocsDomaine.Add(new BlocTravail(blocId, blocDto.Nom, capacite, tachesPourCeBloc));
98 }
99
100 // 6. Mapper Lots
101 var lotsDomaine = new List<LotTravaux>();
102 foreach (var lotDto in inputDto.Lots)
103 {
104 var lotId = new LotId(lotDto.LotId);
105 var blocIdsPourCeLot = lotDto.BlocIds.Select(id => new BlocId(id)).ToList();
106 // Ici, on suppose que les BlocId dans lotDto.BlocIds existent bien.
107 // La validation des références croisées se fera plus tard par IChantierValidationService
108 // ou directement par le constructeur de Chantier si on y déplace cette logique.
109
110 lotsDomaine.Add(new LotTravaux(
111 lotId,
112 lotDto.Nom,
113 lotDto.Priorite,
114 blocIdsPourCeLot,
115 lotDto.DateDebutAuPlusTotSouhaitee,
116 lotDto.DateFinAuPlusTardSouhaitee));
117 }
118
119 // 7. Mapper ConfigurationCdC (si présente)
120 ConfigurationChantier? configCdCDomaine = null;
121 if (inputDto.ConfigurationCdC != null)
122 {
123 configCdCDomaine = ConfigurationChantier.CreerDepuisIds(
124 inputDto.ConfigurationCdC.OuvriersClefsIds,
125 inputDto.ConfigurationCdC.MetiersClefsIds);
126 }
127
128 // 8. Créer l'agrégat Chantier
129 // C'est le constructeur de Chantier qui effectuera les validations de cohérence d'ensemble
130 // (ex: un bloc dans un lot existe, une tâche référence un métier existant, etc.)
131 chantierDomaine = new Chantier(
132 chantierId,
133 inputDto.Description,
134 periodeSouhaitee,
135 calendrierOuvre, // Cet objet est crucial
136 metiersDomaine,
137 ouvriersDomaine,
138 blocsDomaine,
139 lotsDomaine,
140 configCdCDomaine);
141 }
142 catch (ArgumentException ex) // Capturer les exceptions de validation des VOs/Entités
143 {
144 messages.Add(new MessageValidationDto
145 {
146 Type = TypeMessageValidation.Erreur,
147 CodeMessage = "ERR_MAP_ARGUMENT", // Code générique pour erreur d'argument au mapping
148 Message = $"Donnée invalide : {ex.Message}",
149 ProprieteConcernee = ex.ParamName // Nom du paramètre qui a causé l'erreur
150 });
151 // chantierDomaine restera null ou sera l'état partiel avant l'exception
152 }
153 catch (InvalidOperationException ex) // Capturer les exceptions d'opération invalide (ex: ID dupliqué)
154 {
155 messages.Add(new MessageValidationDto
156 {
157 Type = TypeMessageValidation.Erreur,
158 CodeMessage = "ERR_MAP_INVALID_OP",
159 Message = $"Opération invalide lors du mapping : {ex.Message}"
160 });
161 }
162 // On pourrait ajouter d'autres catch plus spécifiques si nécessaire
163
164 return Task.FromResult((chantierDomaine, messages));
165 }
166 }
